add_namespace = ideology

country_event = {
    id = ideology.1
    title = ideology.1.t
    desc = ideology.1.d
    picture = GFX_report_event_generic_order
    
    fire_only_once = yes
	
	trigger = {
		AND = {
			NOT = { has_government = order }
			NOT = { has_government = fanaticism }
			ideology = { order > 0.89 ]
		}
	}
    
    option = {
        name = ideology.1.a
		set_politics = { ruling_party = order elections_allowed = no }
    }
	
	option = {
        name = ideology.1.b
		add_popularity = {
			ideology = order
			popularity = -0.5
		}
		start_civil_war = {
			ideology = order
			size = 0.8
		}
    }
}
